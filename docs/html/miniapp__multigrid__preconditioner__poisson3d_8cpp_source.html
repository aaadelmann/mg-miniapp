1// Mini-app: Matrix-Free Multigrid-Preconditioned CG for 3D Poisson (independent nx,ny,nz)
2// -----------------------------------------------------------------------------
3// Serial C++17, single-file, educational. No deps beyond the STL.
4// Domain: [0,1]^3 with homogeneous Dirichlet BCs (u=0 on boundary).
5// Discretization: 7-point FD Laplacian, allowing hx!=hy!=hz.
6// Hierarchy: geometric 2:1 coarsening per-dimension until coarsest <= 3^3.
7// Smoother: weighted Jacobi implemented as u <- u + Ï‰ D^{-1} (f - A u).
8// Transfers: full-weighting restriction (3D), trilinear prolongation.
9// Preconditioner: one V-cycle used in right-preconditioned CG (PCG).
10// Matrix-free: no sparse matrices are assembled; A*v is applied on-the-fly.
11// Manufactured test: u=sin(pi x) sin(pi y) sin(pi z), f=3 pi^2 u.
12//
13// Build:
14// g++ -O3 -std=c++17 miniapp_multigrid_preconditioner_poisson3d.cpp -o mg3d_pcg
15// Run:
16// ./mg3d_pcg -nx 65 -ny 97 -nz 129 -nu1 2 -nu2 2 -w 0.8 -tol 1e-8 -maxit 200
17// -----------------------------------------------------------------------------

116 void build_hierarchy(int nx_f, int ny_f, int nz_f) {
117 int nx = nx_f, ny = ny_f, nz = nz_f;
118 if (nx < 3 || ny < 3 || nz < 3) { std::cerr << "Grid too small. Use dims >= 3.\n"; std::exit(1);}
119 while (true) {
120 Level3D lev; lev.nx = nx; lev.ny = ny; lev.nz = nz;
121 lev.hx = 1.0 / (nx - 1); lev.hy = 1.0 / (ny - 1); lev.hz = 1.0 / (nz - 1);
122 L.push_back(lev);
123 if (nx <= 3 || ny <= 3 || nz <= 3) break;
124 auto half = [](int n){ return (n-1)/2 + 1; };
125 nx = std::max(3, half(nx));
126 ny = std::max(3, half(ny));
127 nz = std::max(3, half(nz));
128 }

150 static void apply_A(const Level3D& lev, const std::vector<double>& u, std::vector<double>& out) {
151 const int nx = lev.nx, ny = lev.ny, nz = lev.nz;
152 const double ihx2 = 1.0/(lev.hx*lev.hx);
153 const double ihy2 = 1.0/(lev.hy*lev.hy);
154 const double ihz2 = 1.0/(lev.hz*lev.hz);
155 const double diag = 2.0*ihx2 + 2.0*ihy2 + 2.0*ihz2;
156 std::fill(out.begin(), out.end(), 0.0);
157 for (int k=1;k<nz-1;++k)
158 for (int j=1;j<ny-1;++j)
159 for (int i=1;i<nx-1;++i) {
160 const int id = id3(i,j,k,nx,ny);
161 out[id] = diag * u[id]
162 - ihx2*(u[id3(i-1,j,k,nx,ny)] + u[id3(i+1,j,k,nx,ny)])
163 - ihy2*(u[id3(i,j-1,k,nx,ny)] + u[id3(i,j+1,k,nx,ny)])
164 - ihz2*(u[id3(i,j,k-1,nx,ny)] + u[id3(i,j,k+1,nx,ny)]);
165 }
166 }

195 static void smooth_jacobi(const Level3D& lev, std::vector<double>& u, const std::vector<double>& f,
196 std::vector<double>& r_scratch, std::vector<double>& Au_scratch,
197 int iters, double omega) {
198 const int nx=lev.nx, ny=lev.ny, nz=lev.nz;
199 const double ihx2 = 1.0/(lev.hx*lev.hx);
200 const double ihy2 = 1.0/(lev.hy*lev.hy);
201 const double ihz2 = 1.0/(lev.hz*lev.hz);
202 const double D = 2.0*ihx2 + 2.0*ihy2 + 2.0*ihz2; // diag(A)
203 for (int it=0; it<iters; ++it) {
204 residual(lev, u, f, r_scratch, Au_scratch);
205 for (int k=1;k<nz-1;++k)
206 for (int j=1;j<ny-1;++j)
207 for (int i=1;i<nx-1;++i) {
208 const int id = id3(i,j,k,nx,ny);
209 u[id] += omega * (r_scratch[id] / D);
210 }
211 }
212 }

221 static void restrict_fullweight(const Level3D& fine, const std::vector<double>& r_f,
222 const Level3D& coarse, std::vector<double>& f_c) {
223 const int nxf=fine.nx, nyf=fine.ny, nzf=fine.nz;
224 const int nxc=coarse.nx, nyc=coarse.ny, nzc=coarse.nz;
225 std::fill(f_c.begin(), f_c.end(), 0.0);
226 for (int K=1; K< nzc-1; ++K)
227 for (int J=1; J< nyc-1; ++J)
228 for (int I=1; I< nxc-1; ++I) {
229 const int i = 2*I, j = 2*J, k = 2*K;
230 double sum = 0.0;
231 for (int dk=-1; dk<=1; ++dk)
232 for (int dj=-1; dj<=1; ++dj)
233 for (int di=-1; di<=1; ++di) {
234 int wclass = std::abs(di) + std::abs(dj) + std::abs(dk);
235 double w = (wclass==0)?8.0 : (wclass==1?4.0 : (wclass==2?2.0:1.0));
236 sum += w * r_f[id3(i+di, j+dj, k+dk, nxf, nyf)];
237 }
238 f_c[id3(I,J,K,nxc,nyc)] = sum / 64.0;
239 }

248 static void prolong_add(const Level3D& coarse, const std::vector<double>& u_c,
249 const Level3D& fine, std::vector<double>& u_f) {
250 const int nxc=coarse.nx, nyc=coarse.ny, nzc=coarse.nz;
251 const int nxf=fine.nx, nyf=fine.ny, nzf=fine.nz;
252 for (int K=1; K< nzc-1; ++K)
253 for (int J=1; J< nyc-1; ++J)
254 for (int I=1; I< nxc-1; ++I) {
255 const double c000 = u_c[id3(I, J, K, nxc,nyc)];
256 const double c100 = u_c[id3(I+1,J, K, nxc,nyc)];
257 const double c010 = u_c[id3(I, J+1,K, nxc,nyc)];
258 const double c110 = u_c[id3(I+1,J+1,K, nxc,nyc)];
259 const double c001 = u_c[id3(I, J, K+1,nxc,nyc)];
260 const double c101 = u_c[id3(I+1,J, K+1,nxc,nyc)];
261 const double c011 = u_c[id3(I, J+1,K+1,nxc,nyc)];
262 const double c111 = u_c[id3(I+1,J+1,K+1,nxc,nyc)];
263
264 const int i = 2*I, j = 2*J, k = 2*K;
265
266 // layer k
267 u_f[id3(i, j, k, nxf,nyf)] += c000;
268 u_f[id3(i+1,j, k, nxf,nyf)] += 0.5*(c000 + c100);
269 u_f[id3(i, j+1,k, nxf,nyf)] += 0.5*(c000 + c010);
270 u_f[id3(i+1,j+1,k, nxf,nyf)] += 0.25*(c000 + c100 + c010 + c110);
271
272 // layer k+1
273 u_f[id3(i, j, k+1,nxf,nyf)] += 0.5*(c000 + c001);
274 u_f[id3(i+1,j, k+1,nxf,nyf)] += 0.25*(c000 + c100 + c001 + c101);
275 u_f[id3(i, j+1,k+1,nxf,nyf)] += 0.25*(c000 + c010 + c001 + c011);
276 u_f[id3(i+1,j+1,k+1,nxf,nyf)] += 0.125*(c000 + c100 + c010 + c110
277 + c001 + c101 + c011 + c111);
278 }

284 void vcycle(int ell) {
285 Level3D& lev = L[ell];
286 auto &u = uL[ell], &f = fL[ell], &r = rL[ell], &t = tL[ell];
287 if (ell == (int)L.size()-1) {
288 smooth_jacobi(lev, u, f, r, t, 50, omega); // coarsest pseudo-solve
289 return;
290 }
291 smooth_jacobi(lev, u, f, r, t, nu1, omega);
292 residual(lev, u, f, r, t);
293 restrict_fullweight(L[ell], r, L[ell+1], fL[ell+1]);
294 std::fill(uL[ell+1].begin(), uL[ell+1].end(), 0.0);
295 vcycle(ell+1);
296 prolong_add(L[ell+1], uL[ell+1], L[ell], uL[ell]);
297 smooth_jacobi(lev, u, f, r, t, nu2, omega);
298 }

329 static inline double dot(const std::vector<double>& a, const std::vector<double>& b){ double s=0; for(size_t i=0;i<a.size();++i) s+=a[i]*b[i]; return s; }

343 int solve(const Level3D& levA, std::vector<double>& u, const std::vector<double>& f,
344 MultiGrid3D& mg, double& final_relres) {
345 const int N = (int)u.size();
346 std::vector<double> r(N,0.0), z(N,0.0), p(N,0.0), Ap(N,0.0);
347
348 MultiGrid3D::apply_A(levA, u, Ap);
349 for (int i=0;i<N;++i) r[i] = f[i] - Ap[i];
350
351 double normf = nrm2(f); if (normf==0.0) normf=1.0;
352 double rel = nrm2(r)/normf; if (rel < tol) { final_relres=rel; return 0; }
353
354 mg.apply_precond(r, z);
355 p = z; double rz_old = dot(r,z);
356
357 int it=0;
358 for (; it<maxit; ++it) {
359 MultiGrid3D::apply_A(levA, p, Ap);
360 double alpha = rz_old / dot(p,Ap);
361 axpy(u, alpha, p);
362 axpy(r, -alpha, Ap);
363 rel = nrm2(r)/normf; if (rel < tol) { ++it; break; }
364 mg.apply_precond(r, z);
365 double rz_new = dot(r,z);
366 double beta = rz_new / rz_old;
367 for (int i=0;i<N;++i) p[i] = z[i] + beta*p[i];
368 rz_old = rz_new;
369 }
370 final_relres = rel; return it;
371 }
372};

375void fill_manufactured(const Level3D& lev, std::vector<double>& f, std::vector<double>& u_exact) {
376 const double pi = 3.14159265358979323846;
377 const int nx=lev.nx, ny=lev.ny, nz=lev.nz;
378 f.assign(nx*ny*nz, 0.0); u_exact.assign(nx*ny*nz, 0.0);
379 for (int k=0;k<nz;++k){ double z = k*lev.hz;
380 for (int j=0;j<ny;++j){ double y = j*lev.hy;
381 for (int i=0;i<nx;++i){ double x = i*lev.hx;
382 const int id = id3(i,j,k,nx,ny);
383 double uex = std::sin(pi*x)*std::sin(pi*y)*std::sin(pi*z);
384 u_exact[id] = uex;
385 f[id] = 3.0*pi*pi*uex; // -Î” u = (Ï€Â²+Ï€Â²+Ï€Â²) u
386 }
387 }
388 }
389 // enforce zero RHS on Dirichlet boundary nodes
390 for (int k=0;k<nz;++k) for (int j=0;j<ny;++j) { f[id3(0,j,k,nx,ny)]=0; f[id3(nx-1,j,k,nx,ny)]=0; }
391 for (int k=0;k<nz;++k) for (int i=0;i<nx;++i) { f[id3(i,0,k,nx,ny)]=0; f[id3(i,ny-1,k,nx,ny)]=0; }
392 for (int j=0;j<ny;++j) for (int i=0;i<nx;++i) { f[id3(i,j,0,nx,ny)]=0; f[id3(i,j,nz-1,nx,ny)]=0; }
393}
